\begin{frame}{Bối cảnh \& Vấn đề}
    \begin{columns}[T]
        \begin{column}{0.52\textwidth}
            \begin{alertblock}{Hiện trạng}

        \begin{column}{0.44\textwidth}
            \begin{exampleblock}{Giải pháp}
                Xây dựng \highlight{hệ thống CSDL chuyên biệt}:
                \begin{itemize}
                    \item Quản lý nghệ sĩ, hợp đồng, doanh thu
                    \item 3 nguồn thu: Streaming, Sync, Live
                    \item Ví điện tử + rút tiền minh bạch
                    \item Dashboard phân tích real-time
                \end{itemize}
            \end{exampleblock}
        \end{column}
    \end{columns}

    \note{
        Kể câu chuyện: ``Hãy tưởng tượng một label quản lý 50 nghệ sĩ

        từ 3 nguồn thu khác nhau. Sai 1 con số = mất niềm tin nghệ sĩ.''
        Đó là lý do nhóm chọn đề tài này.
    }

% ----- Slide 3: Yêu cầu hệ thống -----
\begin{frame}{Yêu cầu hệ thống}
    \begin{columns}[T]
        \begin{column}{0.50\textwidth}
            \textbf{4 nhóm người dùng:}
            \begin{enumerate}
                \item \textbf{Artist} --- xem doanh thu, ví, rút tiền
                \item \textbf{Label} --- quản lý hợp đồng, chia doanh thu
                \item \textbf{Manager} --- lịch biểu diễn, đặt venue
                \item \textbf{Admin} --- thống kê toàn hệ thống, audit
            \end{enumerate}

            \vspace{0.5em}
            \textbf{Yêu cầu chính:}
            \begin{itemize}\small
                \item Quản lý nghệ sĩ (solo / band / composer)
                \item Hợp đồng + chia tỷ lệ doanh thu
                \item Doanh thu đa nguồn (streaming, sync, live)
                \item Ví điện tử + rút tiền
            \end{itemize}
        \end{column}
        \begin{column}{0.46\textwidth}
            \begin{block}{Business Rules}
                \begin{itemize}\small
                    \item \textbf{BR-01:} Tổng \tbl{share\_percentage} mỗi
                          (contract, track) $\leq$ 1.0
                    \item \textbf{BR-02:} Wallet chỉ giảm khi withdrawal
                          \sql{status = 'completed'}
                    \item \textbf{BR-03:} Revenue logs \coral{append-only}
                          --- sửa bằng reversal entry
                    \item \textbf{BR-05:} Live revenue \textbf{bắt buộc}
                          liên kết \tbl{event\_id}
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}

    \note{
        Nhấn mạnh tính thực tế: ``Đây không phải bài tập giả định ---
        dữ liệu lấy từ MusicBrainz API và Ticketbox thực tế.''
        Business rules là yếu tố quan trọng nhất --- tất cả SP đều
        phải tuân thủ các rules này.
    }
\end{frame}
