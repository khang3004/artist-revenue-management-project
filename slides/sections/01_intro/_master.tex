\section{Introduction}

\begin{frame}{Database Objectives}
    \sectionCard{Artist Revenue Management System}{Database Design Project}

    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \textbf{Problem Statement:}
            \begin{itemize}
                \item Music artists have revenue from \highlight{multiple sources}: streaming, download, live shows
                \item Need to manage \highlight{contracts}, revenue splitting between multiple parties
                \item Track tracks, albums, bookings and payments
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \textbf{Database Goals:}
            \begin{itemize}
                \item Manage artist info, record labels, albums, tracks
                \item Record revenue from multiple sources
                \item Manage contracts \& profit distribution
                \item Statistics, revenue trend analysis
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}
