% ----- Frame: Objectives -----
\begin{frame}{System Objectives}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \begin{block}{Problem Statement}
                Build a \highlight{V-Pop music royalty \& revenue management} system
                covering the full artist lifecycle: contracts, music catalog,
                multi-source revenue streams, and transparent financial settlement
                via digital wallets.
            \end{block}
            \vspace{0.4em}
            \textbf{Three main revenue streams:}
            \begin{itemize}
                \item \textbf{Streaming} --- Spotify, Apple Music, Zing MP3
                \item \textbf{Sync Licensing} --- films, ads, games
                \item \textbf{Live Performance} --- concerts \& shows
            \end{itemize}
        \end{column}
        \begin{column}{0.42\textwidth}
            \begin{exampleblock}{Core Requirements}
                \begin{itemize}\small
                    \item Contract management \& revenue splitting by percentage
                    \item Digital wallets + financial audit trail
                    \item Real-time analytics dashboard
                    \item Automated ETL pipeline from multiple data sources
                    \item Data integrity (append-only revenue logs)
                \end{itemize}
            \end{exampleblock}
        \end{column}
    \end{columns}
\end{frame}

% ----- Frame: User Personas -----
\begin{frame}{User Personas}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{block}{\small Artist}
                \scriptsize View personal revenue, track albums/tracks,
                check wallet balance, view event schedule, request withdrawals
            \end{block}
            \vspace{0.4em}
            \begin{block}{\small Record Label}
                \scriptsize Manage signed artists, create \& manage contracts,
                distribute revenue via contract splits, monitor revenue logs
            \end{block}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{block}{\small Manager / Booking Agent}
                \scriptsize Manage performance schedules, book venues,
                track event status
            \end{block}
            \vspace{0.4em}
            \begin{block}{\small System Administrator}
                \scriptsize System-wide statistics, consolidated revenue reports,
                analytics dashboard, wallet audit
            \end{block}
        \end{column}
    \end{columns}
\end{frame}
