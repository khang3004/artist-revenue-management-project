% ----- Frame: Physical Design -----
\begin{frame}[fragile]{Physical Design --- Highlights}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \begin{lstlisting}[caption={artists --- JSONB metadata}]
CREATE TABLE artists (
  artist_id  SERIAL PRIMARY KEY,
  stage_name VARCHAR(100) UNIQUE NOT NULL,
  full_name  VARCHAR(200),
  debut_date DATE,
  birthday   DATE,
  label_id   INT REFERENCES labels,
  metadata   JSONB  -- genre, social_links
);
            \end{lstlisting}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{lstlisting}[caption={revenue\_logs --- partitioned}]
CREATE TABLE revenue_logs (
  log_id       BIGSERIAL,
  track_id     INT REFERENCES tracks,
  amount       DECIMAL(15,4) NOT NULL,
  currency     VARCHAR(3) DEFAULT 'VND',
  log_date     DATE NOT NULL,
  revenue_type VARCHAR(20),
  raw_data     JSONB,
  PRIMARY KEY (log_id, log_date)
) PARTITION BY RANGE (log_date);
            \end{lstlisting}
        \end{column}
    \end{columns}
    \vspace{0.2em}
    \centering \scriptsize
    Quarterly partitions: \tbl{revenue\_logs\_2025\_q1}, \tbl{revenue\_logs\_2025\_q2}, \ldots

% ----- Frame: Indexes + Views -----
\begin{frame}{Indexes, Views \& Materialized Views}
    \begin{columns}[T]
        \begin{column}{0.50\textwidth}
            \textbf{Key indexes:}
            \begin{itemize}\scriptsize
                \item \tbl{idx\_artists\_metadata\_gin} ---
                      \textbf{GIN} on \tbl{artists.metadata} JSONB,
                      supports \sql{@>} containment operator
                \item \tbl{idx\_revlog\_track\_date} ---
                      Composite B-Tree (\tbl{track\_id}, \tbl{log\_date}),
                      enables partition pruning
                \item \tbl{idx\_splits\_contract\_track} ---
                      Composite for contract\_splits JOIN
                \item \tbl{idx\_contracts\_active} ---
                      \textbf{Partial} index (\texttt{status = 'active'})
                \item \tbl{idx\_artists\_name\_lower} ---
                      \textbf{Expression} index on \sql{LOWER(stage\_name)},
                      supports case-insensitive ILIKE search
            \end{itemize}
        \end{column}
        \begin{column}{0.46\textwidth}
            \textbf{Views (real-time):}
            \begin{itemize}\scriptsize
                \item \tbl{v\_artist\_revenue\_summary}
                \item \tbl{v\_contract\_split\_detail}
            \end{itemize}
            \vspace{0.4em}
            \textbf{Materialized Views (cached):}

                \item \tbl{mv\_top\_tracks\_cached}

            \end{itemize}
