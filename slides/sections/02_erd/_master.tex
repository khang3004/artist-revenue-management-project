% ----- Frame: ISA Relationships -----
\begin{frame}{ISA Relationships --- 3 Inheritance Hierarchies}
    \begin{columns}[T]
        \begin{column}{0.32\textwidth}
            \begin{alertblock}{\small Artists ISA}
                \centering \textbf{Overlapping, Partial}
                \vspace{0.3em}

                \scriptsize
                An artist may hold \textbf{multiple roles simultaneously}
                or belong to \textbf{none} (e.g.\ unclassified DJ)
                \vspace{0.3em}

                $\downarrow$ \tbl{solo\_artists} \\
                $\downarrow$ \tbl{bands} \\
                $\downarrow$ \tbl{composers}
            \end{alertblock}
        \end{column}
        \begin{column}{0.32\textwidth}
            \begin{block}{\small Contracts ISA}
                \centering \textbf{Disjoint, Total}
                \vspace{0.3em}

                \scriptsize
                Every contract \textbf{must} belong to exactly one type --- no overlap, no exceptions
                \vspace{0.3em}

                $\downarrow$ \tbl{recording\_contracts} \\
                $\downarrow$ \tbl{distribution\_contracts} \\
                $\downarrow$ \tbl{publishing\_contracts}
            \end{block}
        \end{column}
        \begin{column}{0.32\textwidth}
            \begin{block}{\small Revenue Logs ISA}
                \centering \textbf{Disjoint, Total}
                \vspace{0.3em}

                \scriptsize
                Every revenue entry belongs to exactly one source --- mandatory classification
                \vspace{0.3em}

                $\downarrow$ \tbl{streaming\_revenue\_details} \\
                $\downarrow$ \tbl{sync\_revenue\_details} \\
                $\downarrow$ \tbl{live\_revenue\_details}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

% ----- Frame: Constraints -----
\begin{frame}{Integrity Constraints \& Business Rules}
    \begin{columns}[T]
        \begin{column}{0.50\textwidth}
            \textbf{Key technical constraints:}
            \begin{itemize}\small
                \item \tbl{tracks.isrc} --- \sql{UNIQUE NOT NULL VARCHAR(12)},
                      international standard tracking code
                \item \tbl{contracts.contract\_id} --- \sql{UUID} PK,
                      prevents ID enumeration attacks
                \item \tbl{artist\_wallets.balance} --- \sql{CHECK >= 0},
                      balance never goes negative
                \item \tbl{revenue\_logs} --- \textbf{Append-only},
                      no \sql{UPDATE} or \sql{DELETE}
                \item \tbl{contract\_splits} --- \sql{CHECK XOR(artist\_id, label\_id)},
                      exactly one beneficiary
                \item \tbl{bands.member\_count} --- \sql{CHECK >= 2}
            \end{itemize}
        \end{column}
        \begin{column}{0.46\textwidth}
            \textbf{Business rules:}
            \begin{itemize}\small
                \item \textbf{BR-01:} Total \tbl{share\_percentage}
                      per (contract, track) $\leq$ 1.0 (100\%)
                \item \textbf{BR-02:} Wallet balance only decreases when
                      withdrawal \sql{status = 'completed'}
                \item \textbf{BR-03:} Revenue logs are append-only ---
                      corrections use a negative-amount reversal entry
                \item \textbf{BR-05:} Live revenue \textbf{must} link
                      to a valid \tbl{event\_id}
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}
