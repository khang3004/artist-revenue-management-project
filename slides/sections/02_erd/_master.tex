    \note{
        Giải thích từng ISA:
        (1) Artists: overlapping vì Sơn Tùng vừa ca sĩ vừa nhạc sĩ.
        (2) Contracts: disjoint vì 1 hợp đồng chỉ thuộc 1 loại.
        (3) Revenue: disjoint vì 1 dòng doanh thu chỉ từ 1 nguồn.
        (4) Beneficiaries: disjoint --- thay polymorphic FK bằng ISA sạch hơn.
        Nếu ERD quá nhỏ trên màn chiếu, zoom vào từng nhóm khi Q\&A.
    }
\end{frame}

\begin{frame}{ISA \& Weak Entities}
    \begin{columns}[T]

            \vspace{0.3em}

            \begin{adjustbox}{max width=\textwidth}
            \begin{tabular}{@{}llll@{}}
                \toprule
                \textbf{Parent}       & \textbf{Children}            & \textbf{Overlap} & \textbf{Total} \\
                \midrule
                \tbl{artists}         & solo, bands, composers       & Overlapping      & Partial  \\
                \tbl{contracts}       & recording, distrib., publish.& Disjoint         & Total    \\
                \tbl{revenue\_logs}   & streaming, sync, live        & Disjoint         & Total    \\
                \tbl{beneficiaries}   & artist\_ben., label\_ben.    & Disjoint         & Total    \\
                \bottomrule
            \end{tabular}
            \end{adjustbox}

            \vspace{0.6em}
            \coral{Beneficiaries ISA} thay thế polymorphic FK \\
            {\scriptsize \tbl{contract\_splits} $\to$ \tbl{beneficiaries} (FK sạch, chuẩn 3NF)}
        \end{column}
        \begin{column}{0.42\textwidth}
            \begin{block}{12 Weak Entities}
                \begin{itemize}\scriptsize
                    \item 9 ISA subtypes \\
                          {\tiny (3 artist + 3 contract + 3 revenue)}
                    \item 2 beneficiary subtypes
                    \item \tbl{artist\_wallets} (1:1 với artist)
                \end{itemize}
            \end{block}

            \vspace{0.3em}
            \begin{exampleblock}{\scriptsize Ví dụ ISA overlapping}
                \scriptsize
                \textbf{Sơn Tùng M-TP} xuất hiện trong \textit{cả}
                \tbl{solo\_artists} (vocal) \textit{và}
                \tbl{composers} (sáng tác)
            \end{exampleblock}
        \end{column}
    \end{columns}

    \note{
        ``ISA overlapping cho Artists vì Sơn Tùng vừa ca sĩ vừa nhạc sĩ
        --- nếu dùng disjoint sẽ mất thông tin vai trò kép.''
        Beneficiaries ISA: trước đây contract\_splits dùng polymorphic FK
        (artist\_id XOR label\_id). Nhóm refactor sang ISA
        để FK luôn trỏ đúng 1 bảng --- sạch hơn, dễ JOIN hơn.
    }
\end{frame}
