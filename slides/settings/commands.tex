% --- Color Definitions ---
\definecolor{techBlue}{RGB}{0, 85, 150}
\definecolor{alertRed}{RGB}{200, 30, 30}
\definecolor{codeGreen}{RGB}{0, 150, 80}
\definecolor{DeepBlue}{RGB}{0, 51, 102}
\definecolor{goldAccent}{RGB}{180, 140, 20}
\definecolor{lightGray}{RGB}{240, 240, 240}

% --- Text Highlighting ---
\newcommand{\highlight}[1]{\textbf{\textcolor{techBlue}{#1}}}
\newcommand{\keyterm}[1]{\textbf{\textcolor{brown}{#1}}}
\newcommand{\todo}[1]{\textcolor{alertRed}{\textbf{[TODO: #1]}}}

% --- Table Name Styling ---
\newcommand{\tbl}[1]{\texttt{\textcolor{techBlue}{#1}}}

% --- SQL Keyword ---
\newcommand{\sql}[1]{\texttt{\textcolor{codeGreen}{\small #1}}}

% --- Metric Result Box ---
\newcommand{\resultBox}[2]{
    \begin{beamercolorbox}[colsep*=1ex, rounded=true, shadow=false, center]{block title}
        \small #1 \\
        \LARGE \textbf{#2}
    \end{beamercolorbox}
}

% --- KPI Card ---
\newcommand{\kpiCard}[3]{
    \begin{beamercolorbox}[sep=0.5em, wd=#1, shadow=true, rounded=true, center]{block title}
        \small #2 \\
        \Large \textbf{\textcolor{techBlue}{#3}}
    \end{beamercolorbox}
}

% --- Section Header Card ---
\newcommand{\sectionCard}[2]{
    \begin{beamercolorbox}[sep=8pt, center, shadow=true, rounded=true]{title}
        \usebeamerfont{title}#1\par%
        \vskip0.25em%
        {\usebeamerfont{subtitle}\small \textit{#2}}
    \end{beamercolorbox}
    \vspace{0.5cm}
}
